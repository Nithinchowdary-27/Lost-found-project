@app.route("/register", methods=["GET", "POST"]) def register(): if request.method == "POST": username = request.form["username"] password = request.form["password"] conn = sqlite3.connect("/tmp/database.db") c = conn.cursor() c.execute("INSERT INTO users (username, password) VALUES (?, ?)", (username, password)) conn.commit() conn.close() print("User registered:", username) # DEBUG return redirect("/login") return render_template("register.html")
